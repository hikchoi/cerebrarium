pattern = rf"""
^[{prefix}]?     # optionally match prefix at start

    (?P<left>    # capture named group left
        \d+?     # lazily match digits
    )
    {infix}      # match infix
    (?P<right>   # capture named group right
        \d+      # match digits
    )
)
[{suffix}]?$     # optionally match suffix at start
"""
